var newContextCreated = false
var num = 0

// Configuración del SteamingContext y definición del procesamiento.
def creatingFunc(): StreamingContext = {
  val ssc = new StreamingContext(sc, slideInterval)
  val auth = Some(new OAuthAuthorization(new ConfigurationBuilder().build()))
 
  // val twitterStream = TwitterUtils.createStream(ssc, auth) // Si no queremos filtrar por localización
  val locationsQuery = new FilterQuery().locations(boundingBoxes : _*)
  val twitterStream = TwitterUtils.createFilteredStream(ssc, auth, Some(locationsQuery))

  // Parseamos los tweets para quedarnos solo con los hashtags
  val hashTagStream = twitterStream.map(_.getText).flatMap(_.split(" ")).filter(_.startsWith("#"))

  hashTagStream.print()
  
  newContextCreated = true
  ssc
}
